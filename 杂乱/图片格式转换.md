# 现代图片格式转换器：一键将 JPG/PNG 转换为 PSD，保持文件夹结构

## 工具功能亮点

1. **批量转换**：支持将指定目录下的所有 JPG/PNG 图片批量转换为 PSD 格式。
2. **保持文件夹结构**：转换后的文件会按照源目录的文件夹结构保存，避免文件混乱。
3. **树状图预览**：在转换前，用户可以预览源目录和目标目录的文件结构，确保转换后的结果符合预期。
4. **进度显示**：转换过程中会实时显示进度条和当前处理的文件，方便用户了解转换进度。
5. **错误处理**：如果转换过程中出现错误，工具会捕获并显示错误信息，同时记录失败的文件数量。

## 技术实现

### 1. 使用 `photoshop-python-api` 进行文件转换

工具的核心功能是通过 `photoshop-python-api` 库与 Photoshop 进行交互，实现图片格式的转换。具体步骤如下：

- 打开源文件（JPG/PNG）。
- 将文件另存为 PSD 格式。
- 关闭文档。

```python
def convert_to_psd(source_file, target_file):
    with Session() as ps:
        ps.app.open(source_file)
        doc = ps.active_document
        options = ps.PhotoshopSaveOptions()
        doc.saveAs(target_file, options, True)
        doc.close()
```

### 2. 树状图展示文件结构

为了直观地展示源目录和目标目录的文件结构，工具使用了 `ttk.Treeview` 组件。树状图可以动态更新，用户选择目录后，文件结构会立即显示在界面中。

```python
def _build_tree(self, tree, directory):
    """递归构建树状图"""
    if not os.path.exists(directory):
        return
    
    root_node = tree.insert("", "end", text=directory, open=True)
    for root, dirs, files in os.walk(directory):
        relative_path = os.path.relpath(root, directory)
        parent_node = root_node
        if relative_path != ".":
            parent_node = tree.insert(root_node, "end", text=relative_path, open=True)
        
        for file in files:
            if file.lower().endswith(('.jpg', '.jpeg', '.png')):
                tree.insert(parent_node, "end", text=file)
```

## 使用步骤

1. **选择源目录**：点击“选择目录”按钮，选择包含 JPG/PNG 图片的文件夹。
2. **选择目标目录**：点击“选择目录”按钮，选择保存 PSD 文件的文件夹。
3. **预览文件结构**：点击“预览文件”按钮，查看源目录和目标目录的文件结构。
4. **开始转换**：点击“开始转换”按钮，工具会自动将图片转换为 PSD 格式，并保存到目标目录中。

## 完整代码

```python
import os
import tkinter as tk
from tkinter import filedialog, messagebox, ttk
from tkinter.scrolledtext import ScrolledText
from photoshop import Session
import time

class ModernImageConverter:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("现代图片格式转换器")
        self.window.geometry("1000x700")
        self.window.configure(bg="white")
        
        # 设置全局样式
        style = ttk.Style()
        style.configure("TFrame", background="white")
        style.configure("TLabelframe", background="white", borderwidth=2)
        style.configure("TLabelframe.Label", background="white", font=("微软雅黑", 10))
        style.configure("TButton", 
                       padding=10,
                       background="white",
                       borderwidth=0,
                       font=("微软雅黑", 9))
        style.configure("Rounded.TButton",
                       padding=10,
                       relief="flat",
                       background="#4a90e2",
                       foreground="white",
                       borderwidth=0,
                       font=("微软雅黑", 9, "bold"))
        style.configure("TEntry", 
                       padding=8,
                       relief="flat",
                       borderwidth=0)
        style.configure("TProgressbar",
                       thickness=10,
                       background="#4a90e2",
                       troughcolor="#f5f5f5",
                       borderwidth=0)
        
        # 创建主容器
        self.container = ttk.Frame(self.window, padding="20")
        self.container.pack(fill=tk.BOTH, expand=True)
        
        # 标题
        title_label = ttk.Label(
            self.container,
            text="图片格式转换器",
            font=("微软雅黑", 18, "bold"),
            background="white"
        )
        title_label.pack(pady=(0, 20))
        
        # 源目录选择框
        self.source_frame = ttk.LabelFrame(
            self.container,
            text="源目录",
            padding="15"
        )
        self.source_frame.pack(fill=tk.X, pady=(0, 10))
        
        self.source_path = tk.StringVar()
        source_entry = ttk.Entry(
            self.source_frame,
            textvariable=self.source_path,
            style="TEntry"
        )
        source_entry.pack(side=tk.LEFT, fill=tk.X, expand=True, padx=(0, 10))
        
        source_button = ttk.Button(
            self.source_frame,
            text="选择目录",
            command=self.select_source,
            style="Rounded.TButton"
        )
        source_button.pack(side=tk.RIGHT)
        
        # 目标目录选择框
        self.target_frame = ttk.LabelFrame(
            self.container,
            text="目标目录", 
            padding="15"
        )
        self.target_frame.pack(fill=tk.X, pady=(0, 10))
        
        self.target_path = tk.StringVar()
        target_entry = ttk.Entry(
            self.target_frame,
            textvariable=self.target_path,
            style="TEntry"
        )
        target_entry.pack(side=tk.LEFT, fill=tk.X, expand=True, padx=(0, 10))
        
        target_button = ttk.Button(
            self.target_frame,
            text="选择目录",
            command=self.select_target,
            style="Rounded.TButton"
        )
        target_button.pack(side=tk.RIGHT)
        
        # 预览区域
        self.preview_frame = ttk.LabelFrame(
            self.container,
            text="文件预览",
            padding="15"
        )
        self.preview_frame.pack(fill=tk.BOTH, expand=True, pady=(0, 10))
        
        # 创建左右两个 Treeview
        self.tree_frame = ttk.Frame(self.preview_frame)
        self.tree_frame.pack(fill=tk.BOTH, expand=True)
        
        # 源目录树状图
        self.source_tree = ttk.Treeview(
            self.tree_frame,
            columns=("type"),
            show="tree",
            selectmode="browse"
        )
        self.source_tree.heading("#0", text="源目录结构", anchor="w")
        self.source_tree.column("#0", stretch=True)
        self.source_tree.pack(side=tk.LEFT, fill=tk.BOTH, expand=True, padx=(0, 10))
        
        # 目标目录树状图
        self.target_tree = ttk.Treeview(
            self.tree_frame,
            columns=("type"),
            show="tree",
            selectmode="browse"
        )
        self.target_tree.heading("#0", text="目标目录结构", anchor="w")
        self.target_tree.column("#0", stretch=True)
        self.target_tree.pack(side=tk.RIGHT, fill=tk.BOTH, expand=True)
        
        # 进度条和状态
        self.progress_var = tk.DoubleVar()
        self.progress_bar = ttk.Progressbar(
            self.container,
            variable=self.progress_var,
            maximum=100,
            style="TProgressbar"
        )
        self.progress_bar.pack(fill=tk.X, pady=(0, 5))
        
        self.status_label = ttk.Label(
            self.container,
            text="准备就绪",
            font=("微软雅黑", 9),
            background="white"
        )
        self.status_label.pack(anchor="w")
        
        # 控制按钮区域
        button_frame = ttk.Frame(self.container)
        button_frame.pack(fill=tk.X, pady=(10, 0))
        
        preview_button = ttk.Button(
            button_frame,
            text="预览文件",
            command=self.preview_conversion,
            style="TButton"
        )
        preview_button.pack(side=tk.LEFT, padx=(0, 10))
        
        convert_button = ttk.Button(
            button_frame,
            text="开始转换",
            command=self.start_conversion,
            style="Rounded.TButton"
        )
        convert_button.pack(side=tk.RIGHT)

    def select_source(self):
        directory = filedialog.askdirectory(title="选择源目录")
        if directory:
            self.source_path.set(directory)
            self.update_source_tree(directory)

    def select_target(self):
        directory = filedialog.askdirectory(title="选择目标目录")
        if directory:
            self.target_path.set(directory)
            self.update_target_tree(directory)

    def update_source_tree(self, directory):
        """更新源目录树状图"""
        self.source_tree.delete(*self.source_tree.get_children())
        self._build_tree(self.source_tree, directory)

    def update_target_tree(self, directory):
        """更新目标目录树状图"""
        self.target_tree.delete(*self.target_tree.get_children())
        self._build_tree(self.target_tree, directory)

    def _build_tree(self, tree, directory):
        """递归构建树状图"""
        if not os.path.exists(directory):
            return
        
        root_node = tree.insert("", "end", text=directory, open=True)
        for root, dirs, files in os.walk(directory):
            relative_path = os.path.relpath(root, directory)
            parent_node = root_node
            if relative_path != ".":
                parent_node = tree.insert(root_node, "end", text=relative_path, open=True)
            
            for file in files:
                if file.lower().endswith(('.jpg', '.jpeg', '.png')):
                    tree.insert(parent_node, "end", text=file)

    def preview_conversion(self):
        """预览转换前后的文件结构"""
        source_dir = self.source_path.get()
        target_dir = self.target_path.get()
        
        if not source_dir:
            messagebox.showwarning("提示", "请先选择源目录！")
            return
        
        self.update_source_tree(source_dir)
        if target_dir:
            self.update_target_tree(target_dir)

    def convert_to_psd(self, source_file, target_file):
        with Session() as ps:
            ps.app.open(source_file)
            doc = ps.active_document
            options = ps.PhotoshopSaveOptions()
            doc.saveAs(target_file, options, True)
            doc.close()

    def start_conversion(self):
        source_dir = self.source_path.get()
        target_dir = self.target_path.get()
        
        if not source_dir or not target_dir:
            messagebox.showwarning("提示", "请选择源目录和目标目录！")
            return
            
        total_files = sum(1 for root, _, files in os.walk(source_dir)
                         for file in files
                         if file.lower().endswith(('.jpg', '.jpeg', '.png')))
        
        if total_files == 0:
            messagebox.showinfo("提示", "没有找到可转换的图片文件！")
            return
            
        converted_count = 0
        failed_count = 0
        
        try:
            for root, dirs, files in os.walk(source_dir):
                relative_path = os.path.relpath(root, source_dir)
                target_subdir = os.path.join(target_dir, relative_path)
                os.makedirs(target_subdir, exist_ok=True)
                
                for file in files:
                    if file.lower().endswith(('.jpg', '.jpeg', '.png')):
                        source_file = os.path.join(root, file)
                        target_file = os.path.join(
                            target_subdir,
                            os.path.splitext(file)[0] + '.psd'
                        )
                        
                        try:
                            self.convert_to_psd(source_file, target_file)
                            converted_count += 1
                            self.status_label.config(text=f"正在处理: {file}")
                        except Exception as e:
                            failed_count += 1
                            print(f"转换失败: {source_file}")
                            print(f"错误信息: {str(e)}")
                        
                        progress = (converted_count + failed_count) / total_files * 100
                        self.progress_var.set(progress)
                        self.window.update()
                        time.sleep(0.5)
            
            messagebox.showinfo(
                "完成",
                f"转换完成！\n成功: {converted_count} 个文件\n失败: {failed_count} 个文件"
            )
            
        except Exception as e:
            messagebox.showerror("错误", f"发生错误：{str(e)}")
        finally:
            self.status_label.config(text="准备就绪")
            self.progress_var.set(0)
            self.update_target_tree(target_dir)  # 更新目标目录树状图

    def run(self):
        self.window.mainloop()

if __name__ == "__main__":
    app = ModernImageConverter()
    app.run()
```
